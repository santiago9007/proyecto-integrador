Lógica de Programación

A continuación, se presenta un ejercicio básico en Java, en el que el usuario debe crear una contraseña y vincularla a su documento de identidad (cédula). Luego se le solicitará que ingrese nuevamente la contraseña. Si los datos coinciden correctamente, se le permitirá acceder a un menú simulado. En caso contrario, se le negará el acceso. 


package com.mycompany.practica;

import java.util.Scanner;

/**
 * @author Santiago Román
 */
public class Practica {

    public static void main(String[] args) {

        // Definición de variables
        int cedula;
        String password;
        String passwordDig;

        Scanner sc = new Scanner(System.in);

        System.out.println("Cree una contraseña:");
        password = sc.nextLine();

        System.out.println("Ingrese la cédula:");
        cedula = sc.nextInt();
        sc.nextLine(); // Limpiar buffer

        System.out.println("Ingrese la contraseña:");
        passwordDig = sc.nextLine();

        if (password.equals(passwordDig)) {
            System.out.println("Tienes acceso al siguiente menú:");
            System.out.println("- Actualizar datos");
            System.out.println("- Ingresar datos");
            System.out.println("- Solicitar cita");
        } else {
            System.out.println("Acceso denegado. Contraseña incorrecta.");
        }
    }
}
