Read

select borrow.id_borrow, typeUser.name, users.name, users.name2, users.lastName, users.lastName2, books.title from borrow
inner join typeUser on borrow.id_typeUser = typeUser.id_typeUser
inner join users on borrow.id_users = users.id_users
inner join books on borrow.id_books = books.id_books;
